# Main Pipeline Using EdgeR incl. Plotting (lattice implementation) incl.
# Plotting (ggplot2 implementation) *In the works* WT, Time Series.  by
# Nathaniel (April 2014)

library(limma)
library(edgeR)
library(reshape2)

## Warning: package 'reshape2' was built under R version 3.0.3

library(lattice)

## Warning: package 'lattice' was built under R version 3.0.3

# library(ggplot2)

# Reset Workspace
rm(list = ls())

# Loads RNA-Seq Raw Fragment Count
raw.reads <- read.table("Data/fragment_counts.txt", header = TRUE, sep = "\t")

# Loads Design Matrix for limma
frag.matrix <- read.table("Data/FragmentMatrix.txt", header = TRUE, sep = "\t")

# Loads Truncated GeneList
gene.list <- read.table("Data/GeneID.txt", header = TRUE)

# Loads crp-S Target List
crp.list <- read.table("Data/crp.txt", header = TRUE)

# Associating GeneID as KEY identifier for all datasets
rownames(gene.list) <- gene.list$ID
rownames(raw.reads) <- gene.list$ID

# Creating Data Subset (with Restructuring) Wild Type (K); Time = ALL
small.meta <- subset(frag.matrix, frag.matrix$Type %in% "K" & frag.matrix$Timepoint %in% 
    c("0", "10", "30", "100"))
small.meta$Timepoint <- factor(small.meta$Timepoint)
small.raw <- raw.reads[, colnames(raw.reads) %in% small.meta$Sample]

# Unloading Unneeded Files
rm(raw.reads)
rm(frag.matrix)

# Running Dataset through EdgeR pipeline 1. Creating DGEList from Trimmed
# Data
dge.list <- DGEList(counts = small.raw, group = small.meta$Timepoint)

# 2. Creating Design Matrix
sm.design <- model.matrix(~Timepoint, small.meta)

# 3. Continuing EdgeR Pipeline
dge.common.list <- estimateGLMCommonDisp(dge.list, sm.design)
dge.trend.list <- estimateGLMTrendedDisp(dge.common.list, sm.design)
dge.tag.list <- estimateGLMTagwiseDisp(dge.trend.list, sm.design)

# 4. Fitting and Throwing Out Hits **Current Null Hypothesis: TypeS (i.e.
# WT, time = ALL against 0)
dge.fit <- glmFit(dge.tag.list, sm.design)
dge.lrt <- glmLRT(dge.fit, coef = grep("Timepoint", colnames(coef(dge.fit))))
dge.top <- topTags(dge.lrt, n = Inf)
dge.cpm <- cpm(dge.tag.list)

final.top.table <- dge.top$table[dge.top$table$FDR < 0.05, ]
final.top.cpm <- dge.cpm[rownames(final.top.table), ]
final.top.genes <- gene.list[gene.list$ID %in% rownames(final.top.table), ]

# Define interested gene(s)
interested.gene <- c("HI_0601", "HI_0521", "HI_0520")

# Restructuring data for plotting
reform.meta <- small.meta
rownames(reform.meta) <- small.meta$Sample

interested.norm.counts <- data.frame(Counts = dge.cpm[interested.gene, ])
inc.table.header <- data.frame(t(data.frame(Type = unclass(reform.meta[colnames(dge.cpm), 
    ]$Set), Time = unclass(reform.meta[colnames(dge.cpm), ]$Timepoint))))
colnames(inc.table.header) <- colnames(interested.norm.counts)
interested.norm.counts <- rbind(data.frame(Counts = dge.cpm[interested.gene, 
    ]), inc.table.header)
melted.norm.counts <- data.frame(t(interested.norm.counts))

melted.norm.counts$Time <- as.factor(melted.norm.counts$Time)
levels(melted.norm.counts$Time) <- c("0", "10", "30", "100")
melted.norm.counts$Type <- as.factor(melted.norm.counts$Type)
levels(melted.norm.counts$Type) <- c("A", "B")
melted.norm.counts <- melt(melted.norm.counts, id.vars = c("Type", "Time"), 
    value.name = "normalized.count")

# Plotting using lattice package
xyplot(normalized.count ~ Time | Type, melted.norm.counts, groups = variable, 
    auto.key = TRUE, type = c("l"), grid = TRUE)
